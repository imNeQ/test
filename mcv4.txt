#!/bin/bash

# Konfiguracja
OLD_BUCKET="old-bucket-name"
NEW_BUCKET="new-bucket-name"
MINIO_ALIAS_OLD="old"
MINIO_ALIAS_NEW="new"
FILES_LIST="files_to_migrate.txt"
SUCCESS_LOG="migrated_files.log"
FAILURE_LOG="failed_files.log"
TMP_RETRY_LIST="retry_files.txt"

# Pliki tymczasowe
OLD_LIST="/tmp/old_files_metadata.txt"
NEW_LIST="/tmp/new_files_metadata.txt"
MISSING_FILES="/tmp/missing_files.txt"
MISMATCH_FILES="/tmp/mismatched_files.txt"

# Funkcja: Generowanie listy plików z metadanymi (rozmiar, ETag)
generate_file_list() {
    local alias=$1
    local bucket=$2
    local output_file=$3

    echo "Generowanie listy plików dla ${alias}/${bucket}..."
    mc find "${alias}/${bucket}" --name "*" --exec "mc stat {} | awk '/^Name|Size|ETag/{print \$2}'" > "${output_file}"
}

# Funkcja: Sprawdzanie obecności i zgodności plików
compare_files() {
    echo "Porównywanie plików między bucketami..."
    > "${MISSING_FILES}"
    > "${MISMATCH_FILES}"

    declare -A new_files
    while IFS= read -r file; do
        read -r size
        read -r etag
        new_files["$file"]="$size $etag"
    done < <(paste - - - < "${NEW_LIST}")

    while IFS= read -r file; do
        read -r size
        read -r etag

        if [[ -z "${new_files["$file"]}" ]]; then
            echo "${file}" >> "${MISSING_FILES}"
        else
            target_metadata=(${new_files["$file"]})
            if [[ "${size}" != "${target_metadata[0]}" || "${etag}" != "${target_metadata[1]}" ]]; then
                echo "${file} - Rozmiar lub suma kontrolna niezgodne." >> "${MISMATCH_FILES}"
            fi
        fi
    done < <(paste - - - < "${OLD_LIST}")
}

# Generowanie listy plików
generate_file_list "${MINIO_ALIAS_OLD}" "${OLD_BUCKET}" "${OLD_LIST}"
generate_file_list "${MINIO_ALIAS_NEW}" "${NEW_BUCKET}" "${NEW_LIST}"

# Porównywanie plików
compare_files

# Podsumowanie wyników
if [ -s "${MISSING_FILES}" ]; then
    echo "Pliki brakujące w nowym buckecie (zapisywane w ${FAILURE_LOG}):"
    cat "${MISSING_FILES}" | tee -a "${FAILURE_LOG}"
else
    echo "Nie znaleziono brakujących plików." | tee -a "${SUCCESS_LOG}"
fi

if [ -s "${MISMATCH_FILES}" ]; then
    echo "Pliki niezgodne (rozmiar lub suma kontrolna, zapisywane w ${FAILURE_LOG}):"
    cat "${MISMATCH_FILES}" | tee -a "${FAILURE_LOG}"
else
    echo "Wszystkie pliki mają zgodny rozmiar i sumy kontrolne." | tee -a "${SUCCESS_LOG}"
fi

# Czyszczenie plików tymczasowych
rm -f "${OLD_LIST}" "${NEW_LIST}" "${MISSING_FILES}" "${MISMATCH_FILES}"

echo "Weryfikacja zakończona."
