#!/bin/bash

# Aliasy
stary="stary/bucket"
nowy="nowy/bucket"

# Listy plików
mc ls --recursive $stary > lista_stary.txt
mc ls --recursive $nowy > lista_nowy.txt

# Porównanie list plików
echo "Porównanie list plików..."
diff <(sort lista_stary.txt) <(sort lista_nowy.txt) > plik_diffs.txt
if [ -s plik_diffs.txt ]; then
    echo "Różnice w plikach:"
    cat plik_diffs.txt
else
    echo "Brak różnic w plikach."
fi

# Sprawdzanie sum kontrolnych
echo "Generowanie sum kontrolnych..."
mc find $stary --exec "mc cat {} | md5sum" > checksum_stary.txt
mc find $nowy --exec "mc cat {} | md5sum" > checksum_nowy.txt

echo "Porównanie sum kontrolnych..."
diff checksum_stary.txt checksum_nowy.txt > checksum_diffs.txt
if [ -s checksum_diffs.txt ]; then
    echo "Różnice w sumach kontrolnych:"
    cat checksum_diffs.txt
else
    echo "Wszystkie sumy kontrolne są zgodne."
fi
