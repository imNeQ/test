#!/bin/bash

# Nazwy aliasów i bucketów
OLD_BUCKET_ALIAS="old"
NEW_BUCKET_ALIAS="new"
OLD_BUCKET="${OLD_BUCKET_ALIAS}/your-old-bucket-name"
NEW_BUCKET="${NEW_BUCKET_ALIAS}/your-new-bucket-name"

# Tymczasowe pliki do przechowywania listy plików
OLD_LIST="old_files.txt"
NEW_LIST="new_files.txt"
OLD_CHECKSUMS="old_checksums.txt"
NEW_CHECKSUMS="new_checksums.txt"
MISMATCHED_FILES="mismatched_files.txt"
MISSING_FILES="missing_files.txt"

# Pobierz listę plików z obu bucketów (z rozmiarami)
mc ls --recursive "$OLD_BUCKET" | awk '{print $1, $NF}' > "$OLD_LIST"
mc ls --recursive "$NEW_BUCKET" | awk '{print $1, $NF}' > "$NEW_LIST"

# Generowanie sum kontrolnych dla obu bucketów
while read -r size file; do
    mc cat "$OLD_BUCKET/$file" | md5sum | awk -v f="$file" '{print $1, f}' >> "$OLD_CHECKSUMS"
done < "$OLD_LIST"

while read -r size file; do
    mc cat "$NEW_BUCKET/$file" | md5sum | awk -v f="$file" '{print $1, f}' >> "$NEW_CHECKSUMS"
done < "$NEW_LIST"

# Sprawdź brakujące pliki
comm -23 <(awk '{print $2}' "$OLD_CHECKSUMS" | sort) <(awk '{print $2}' "$NEW_CHECKSUMS" | sort) > "$MISSING_FILES"

# Sprawdź niezgodne pliki (na podstawie sum kontrolnych)
join -j 2 <(sort -k2 "$OLD_CHECKSUMS") <(sort -k2 "$NEW_CHECKSUMS") | awk '$1 != $3 {print $2}' > "$MISMATCHED_FILES"

# Sprawdź wynik
if [ -s "$MISSING_FILES" ]; then
    echo "Niektóre pliki nie zostały przeniesione. Oto lista:"
    cat "$MISSING_FILES"
fi

if [ -s "$MISMATCHED_FILES" ]; then
    echo "Niektóre pliki mają różne zawartości. Oto lista:"
    cat "$MISMATCHED_FILES"
else
    if [ ! -s "$MISSING_FILES" ]; then
        echo "Wszystkie pliki zostały poprawnie przeniesione i zweryfikowane."
    fi
fi

# Wyczyść tymczasowe pliki
rm -f "$OLD_LIST" "$NEW_LIST" "$MISSING_FILES" "$MISMATCHED_FILES" "$OLD_CHECKSUMS" "$NEW_CHECKSUMS"
